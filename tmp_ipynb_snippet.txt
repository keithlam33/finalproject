def get_history_for_symbol(symbol: str, start_date: datetime.date, end_date: datetime.date):
    start_dt = datetime.datetime(start_date.year, start_date.month, start_date.day)
    end_dt = datetime.datetime(end_date.year, end_date.month, end_date.day) + datetime.timedelta(days=1)
    period1 = int(start_dt.timestamp())
    period2 = int(end_dt.timestamp())

    url = f"https://query1.finance.yahoo.com/v8/finance/chart/{symbol}"
    params = {
        "period1": period1,
        "period2": period2,
        "interval": "1d",
        "events": "history",
        "includeAdjustedClose": "true",
    }
    headers = {"User-Agent": "Mozilla/5.0"}

    try:
        resp = requests.get(url, params=params, headers=headers, timeout=20)
        resp.raise_for_status()
        data = resp.json()
        if not data.get("chart") or not data["chart"].get("result"):
            return None
        result = data["chart"]["result"][0]
        timestamps = result.get("timestamp", [])
        if not timestamps:
            return None
        quote = result["indicators"]["quote"][0]

        ts = pd.Series(timestamps, dtype="int64")

        df = pd.DataFrame(
            {
                "symbol": symbol,
                "ts": ts,
                "open": quote.get("open"),
                "high": quote.get("high"),
                "low": quote.get("low"),
                "close": quote.get("close"),
                "volume": quote.get("volume"),
            }
        )
        df["data_type"] = "1d"
        df = df.dropna(subset=["open", "high", "low", "close"])
        if df.empty:
            return None
        return df[["symbol", "data_type", "ts", "open", "high", "low", "close", "volume"]]
    except Exception as e:
        print(f"history fetch error {symbol}: {e}")
        return None


def insert_with_fallback(records, conn, insert_sql):
    try:
        conn.execute(text(insert_sql), records)
        return len(records)
    except Exception as e:
        print(f"[BATCH FAIL] {e}")
        ok = 0
        for row in records:
            try:
                conn.execute(text(insert_sql), row)
                ok += 1
            except Exception as row_err:
                print(f"[ROW FAIL] {row.get('symbol')} {row.get('ts')}: {row_err}")
        return ok


def insert_daily_bulk(df: pd.DataFrame, engine, batch_size: int = 1000):
    if df is None or df.empty:
        return 0
    cols = list(df.columns)
    insert_sql = f"""
    INSERT INTO stock_ohlc ({', '.join(cols)})
    VALUES ({', '.join([f":{c}" for c in cols])})
    ON CONFLICT ON CONSTRAINT uq_symbol_datatype_ts DO NOTHING
    """
    records = df.to_dict(orient="records")
    inserted = 0
    with engine.begin() as conn:
        for i in range(0, len(records), batch_size):
            batch = records[i:i+batch_size]
            inserted += insert_with_fallback(batch, conn, insert_sql)
    return inserted


def initial_history_all(start_date: datetime.date, batch_size=50, per_symbol_sleep=1.0, batch_sleep=30):
    end_date = datetime.date.today() - timedelta(days=1)
    symbols = get_sp500_symbols_from_web()
    engine = get_engine()
    total = 0
    prog_file = "history_progress.txt"
    done = set()
    if os.path.exists(prog_file):
        with open(prog_file, "r") as f: # r -> read
            for line in f:
                done.add(line.strip())

    for i in range(0, len(symbols), batch_size):
        batch = symbols[i : i + batch_size] # start from i to i+batchsize e.g. [0:50]
        for s in batch:
            if s in done:
                print(f"{s} already done, skip")
                continue
            try:
                df = get_history_for_symbol(s, start_date, end_date)
                if df is None or df.empty:
                    print(f"{s} history empty, skip")
                else:
                    n = insert_daily_bulk(df, engine)
                    total += n
                    print(f"{s} inserted {n} rows")
                with open(prog_file, "a") as f:
                    f.write(s + "\n")
                done.add(s)
            except Exception as e:
                print(f"Error processing {s}: {e}")
            time.sleep(per_symbol_sleep)
        print(f"Batch {i//batch_size + 1} done, sleeping {batch_sleep}s") #f"....{表達式}...."　{} 裏面放嘅係「要計出嚟再插入字串」嘅表達式
        time.sleep(batch_sleep)

    print("initial history completed, total rows:", total)
    return total


def backfill_daily_auto():
    engine = get_engine()
    with engine.connect() as conn:
        last = conn.execute(
            text("SELECT MAX(ts) FROM stock_ohlc WHERE data_type='D'")
        ).scalar()  # fetchone() -> one row, scalar()->　only MAX(ts) not whole row

    if last is None:
        start = date(2015, 1, 1)
    else:
        last_date = datetime.datetime.utcfromtimestamp(int(last)).date()
        start = last_date + timedelta(days=1)

    initial_history_all(start)
